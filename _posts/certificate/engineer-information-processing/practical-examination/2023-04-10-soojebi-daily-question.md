  int low, high;
  int mid;
  int data[9] = {1, 2, 3, 4, 5, 6, 7, 8, 9};
  int key = 7;
  int count = 0;

  low = 0;
  high = 8;

  while(low <= high){
	  count++;
	  mid = (low + high) / 2;
	  if(key == data[mid]){
		  printf("%d", count);
		  low = high + 1;
	  }
	  else if(key < data[mid]){
		  high = mid - 1;
	  }
	  else{
		  low = mid + 1;
	  }
  }
}
